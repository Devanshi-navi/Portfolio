    <script>
        function calculateQuote() {
            const serviceRate = parseFloat(document.getElementById('serviceType').value);
            const hours = parseFloat(document.getElementById('hours').value) || 0;
            const travelKms = parseFloat(document.getElementById('travel').value) || 0;
            const postcode = document.getElementById('postcode').value || "Global";

            // Standard NDIS compliant provider travel cost multiplier rules ($0.96 per kilometer margin)
            const baseTotal = serviceRate * hours;
            const travelTotal = travelKms * 0.96;
            const grandTotal = baseTotal + travelTotal;

            // Render calculations into front-facing HTML node states
            document.getElementById('displayTotal').innerText = "$" + grandTotal.toFixed(2);
            document.getElementById('breakdownBase').innerText = "$" + baseTotal.toFixed(2);
            document.getElementById('breakdownTravel').innerText = "$" + travelTotal.toFixed(2);
            document.getElementById('breakdownRegion').innerText = "Postcode " + postcode + " Clear";

            // Live Update State 5 Broadcast parameters dynamically
            document.getElementById('broadcastText').innerText = `Hi [Sitter Name], an NDIS job matching your postcode ${postcode} profile is available paying $${serviceRate.toFixed(2)}/hr. Reply YES to accept.`;
        }

        function updatePersonaHooks() {
            const filter = document.getElementById('personaFilter').value;
            const box = document.getElementById('personaBox');

            const strategies = {
                standard: "Enforces: 100% Police Checks, Vet-endorsed safety credentials, and mandatory live biometric DVS checks.",
                corporate: "Enforces: Priority Last-Minute Booking Routing, Roster Variance Protection, and Automated Pet Taxi Dispatching rules.",
                vulnerable: "Enforces: High-Trust Provider Isolation, Medical Adherence Checklists, and Direct Phone Check-ins.",
                premium: "Enforces: Elite Care matching (RSPCA Collaboration Academy Graduates), Zero Separation configurations."
            };

            box.innerText = strategies[filter] || strategies.standard;
        }
    </script>
